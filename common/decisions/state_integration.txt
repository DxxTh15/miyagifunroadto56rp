GLO_state_integration_category = {
	
	#Toggle to hide all coring
    hide_integration_decisions = {
        priority = 100
        visible = { 
			always = yes 
		}
		available = {
			NOT = { has_country_flag = show_integration_decisions }
		}
        complete_effect = { 
			set_country_flag = show_integration_decisions
        }
    }

	show_integration_decisions = {
        priority = 99
        visible = { 
			always = yes 
		}
		available = {
			has_country_flag = show_integration_decisions 
		}
        complete_effect = { 
            clr_country_flag = show_integration_decisions  
        }
    }

	#Toggle "Show only coring decisions with enough compliance"
	hide_nonavialible_integration_decisions = {
        priority = 98
        visible = { 
			always = yes 
		}
		available = {
			has_country_flag = show_unavialible_integration_decisions
		}
        complete_effect = { 
            clr_country_flag = show_unavialible_integration_decisions # Warlider can't spell for shit - im not even gonna change this.
        }
    }

	show_nonavialible_integration_decisions = {
        priority = 97
        visible = { 
			always = yes 
		}
		available = {
			NOT = { has_country_flag = show_unavialible_integration_decisions }
		}

        complete_effect = { 
            set_country_flag = show_unavialible_integration_decisions
        }
    }

    integrate_the_state_of_FROM_peacefully = {
        icon = GFX_decision_generic_nationalism

        on_map_mode = map_only
        state_target = yes 
        target_array = controlled_states
		
		ai_will_do = {
			base = 0
		}

        target_trigger = {
            FROM = {
                is_fully_controlled_by = ROOT
				NOT = { is_core_of = ROOT }
            }
        }

        available = {
			IF = {
				limit = {
					OR = {
						is_major = yes
						NOT = {
							FROM = {
								is_claimed_by = ROOT
							}
						}
					}
				}
				FROM = {
					compliance > 79
					resistance < 11
				}
				has_manpower > coring_man
				
			}
			ELSE = {
				FROM = {
					compliance > 69
					resistance < 19
				}
				has_manpower > coring_man				
			}
        }

        visible = {
            NOT = { has_country_flag = show_integration_decisions }
			FROM = { 
				NOT = { is_core_of = ROOT } 
				NOT = { has_state_flag = being_cored } 
			}
			IF = {
				limit = {
					is_major = no
				}
				OR = {
					has_country_flag = show_unavialible_integration_decisions
					FROM = {
						compliance > 69
						resistance < 19
					}
				}
			}
			ELSE = {
				OR = {
					has_country_flag = show_unavialible_integration_decisions
					FROM = {
						compliance > 79
						resistance < 11
					}
				}
			}
        }

        custom_cost_trigger = {
			IF = {
				limit = {
					is_major = no
				}
				has_political_power > GLO_p_pp_cost
			}
			ELSE_IF = {
				limit = {
					is_major = yes
				}
				has_political_power > GLO_p_pp_cost
			}
        }
        custom_cost_text = GLO_35_pol_power

        days_remove = 50

        complete_effect = {
			set_temp_variable = { GLO_p_mp_cost_t = GLO_p_mp_cost }
			set_temp_variable = { GLO_p_pp_cost_t = GLO_p_pp_cost }
			multiply_temp_variable = { GLO_p_mp_cost_t = -1 }
			multiply_temp_variable = { GLO_p_pp_cost_t = -1 }
            add_manpower = GLO_p_mp_cost_t
			add_political_power = GLO_p_pp_cost_t
			hidden_effect = {
				FROM = { 
					set_state_flag = being_cored
				}
			}
        }

        remove_effect = {
            FROM = {
                add_core_of = ROOT
				clr_state_flag = being_cored
            }
        }
    }

    integrate_the_state_of_FROM_by_force = {
        icon = GFX_decision_oppression

		on_map_mode = map_only
        state_target = yes 
        target_array = controlled_states

		ai_will_do = {
			base = 0
		}

        target_trigger = {
            FROM = {
                is_fully_controlled_by = ROOT
				NOT = { is_core_of = ROOT }
            }
        }

        available = {
			NOT = {
				OR = {
					has_government = neutrality
					has_government = democratic
				}
			}
			hidden_trigger = {
				FROM = {
					NOT = { is_core_of = ROOT }
					NOT = { has_state_flag = being_cored }
				}
			}
			IF = {
				limit = {
					OR = { #if not a major, or if a major has a claim, discounted coring.
						NOT = { 
							is_major = yes
						}
						FROM = {
							is_claimed_by = ROOT
						}
					}
				}
				FROM = {
					compliance > 59
					resistance < 31
				}
				has_manpower > 12500
				has_equipment = {
					infantry_equipment > FROM.r56rp_core_cost_minor
				}
			}
			ELSE = {
				FROM = {
					compliance > 59
					resistance < 31
				}
				has_manpower > 12500
				has_equipment = {
					infantry_equipment > FROM.r56rp_core_cost
				}
				
			}
        }

        visible = {
            NOT = { 
				has_country_flag = show_integration_decisions 
			}
			OR = {
				has_country_flag = show_unavialible_integration_decisions
				FROM = {
					compliance > 59
					resistance < 31
				}
			}
			NOT = {
				OR = {
					has_government = neutrality
					has_government = democratic
				}
			}
			FROM = {
				NOT = { is_core_of = ROOT }
				NOT = { has_state_flag = being_cored }
			}
        }

		#tried making it display guns but failed miserably. so itll remain on manpower.
        custom_cost_trigger = { 
			has_manpower > 12500
        }
        custom_cost_text = GLO_1000_inf_eqmnt

        days_remove = 75

        complete_effect = {
			IF = {
				limit = {
					OR = { #if not a major, or if a major has a claim, discounted coring.
					NOT = { 
						is_major = yes
					}
					FROM = {
						is_claimed_by = ROOT
					}
				}
				}
				set_temp_variable = { GLO_f_inf_cost_t = FROM.r56rp_core_cost_minor }
				multiply_temp_variable = { GLO_f_inf_cost_t = -1 }
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = GLO_f_inf_cost_t
				}
			}
			ELSE = {
				set_temp_variable = { GLO_f_inf_cost_t = FROM.r56rp_core_cost }
				multiply_temp_variable = { GLO_f_inf_cost_t = -1 }
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = GLO_f_inf_cost_t
				}
			}
			set_temp_variable = { GLO_f_mp_cost_t = GLO_f_mp_cost }
			multiply_temp_variable = { GLO_f_mp_cost_t = -1 }
            add_manpower = GLO_f_mp_cost_t
            
			hidden_effect = {
				FROM = { 
					set_state_flag = being_cored
				}
			}
        }

        remove_effect = {
            FROM = {
                add_core_of = ROOT
				clr_state_flag = being_cored
            }
        }
    }
}
