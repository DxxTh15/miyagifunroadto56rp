DIPLOMACY_CREATE_FACTION_ENABLE_TRIGGER = {
	OR = {
		AND = { # AI can invite ai
			ROOT = { #sender
				is_ai = yes
			}
			FROM = { #recipient
				is_ai = yes
			}
		}
		AND = { # Players can invite players
			ROOT = { #sender
				is_ai = no
			}
			FROM = { #recipient
				is_ai = no
			}
		}
	}
